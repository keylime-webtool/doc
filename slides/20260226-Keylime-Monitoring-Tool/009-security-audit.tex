\begin{frame}{Security Audit - Identity Verification Events}
  \begin{block}{TPM Identity Immutability}
    \scriptsize
    Monitoring re-registration attempts and TPM identity changes is critical for detecting
    potential agent impersonation attacks.
  \end{block}

  \vspace{0.2cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Registration Events:}
      \begin{itemize}
        \item New agent registration (with identity type)
        \item Agent re-registration (regcount tracking)
        \item Identity type used:
        \begin{itemize}
          \scriptsize
          \item \texttt{ek\_cert} - EK certificate
          \item \texttt{iak\_idevid} - IAK/IDevID
          \item \texttt{default} - Basic TPM keys
        \end{itemize}
        \item EK certificate validation results
        \item TPM key changes between registrations
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Identity Alerts:}
      \begin{itemize}
        \item \textcolor{dangered}{CRITICAL}: TPM key changed on re-registration
        \item \textcolor{dangered}{CRITICAL}: EK cert mismatch on re-registration
        \item \textcolor{warningyellow}{WARNING}: High regcount (>3)
        \item \textcolor{warningyellow}{WARNING}: Re-registration from different IP
        \item \textcolor{infoblue}{INFO}: Normal re-registration after reboot
      \end{itemize}

      \vspace{0.2cm}

      \begin{alertblock}{Key Principle}
        \scriptsize
        TPM identity keys should be \textbf{immutable}. Any change in EK/AK between
        registrations indicates potential compromise.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}
