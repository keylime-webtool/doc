  \begin{block}{Attestation Record Backends}
    \scriptsize
    Keylime supports multiple backends for durable attestation recording,
    providing audit trails and evidence preservation.
  \end{block}

  \vspace{0.1cm}

  \scriptsize
  \begin{center}
    \begin{tabular}{|l|p{3cm}|p{2.5cm}|p{2.5cm}|}
      \hline
      \textbf{Backend} & \textbf{Use Case} & \textbf{Monitored Metrics} & \textbf{Alert Conditions} \\
      \hline
      \hline
      Rekor & Transparency log (tamper-evident) & Upload latency, inclusion proofs & Upload failures, log unavailable \\
      \hline
      Redis & Time-series attestation data & Latency, memory usage, key count & Connection lost, high latency \\
      \hline
      SQL DB & Persistent storage & Query time, storage growth & Storage full, slow queries \\
      \hline
      File-based & Local audit trail & Disk usage, write speed & Disk full, permission errors \\
      \hline
      RFC3161 TSA & Timestamping authority & Response time, token validity & TSA unreachable, cert expired \\
      \hline
    \end{tabular}
  \end{center}
