\section{Introduction}

\begin{frame}{Introduction - The Challenge}
  \begin{block}{Remote Attestation at Scale}
    As organizations deploy Keylime across hundreds or thousands of machines,
    system administrators face a critical challenge: \textbf{how to effectively monitor,
    operate, and troubleshoot a distributed trust infrastructure}.
  \end{block}

  \vspace{0.3cm}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Without a Dashboard:}
      \begin{itemize}
        \item Manual CLI queries per agent
        \item No visibility into fleet health
        \item Delayed detection of failures
        \item Complex log correlation
        \item No trend analysis capability
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{With a Dashboard:}
      \begin{itemize}
        \item Real-time fleet overview
        \item Immediate failure alerts
        \item Historical trend analysis
        \item Centralized policy management
        \item Operational efficiency
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Introduction - Why Keylime Monitoring?}
  \begin{alertblock}{Business Case}
    \footnotesize
    In production environments, a single undetected attestation failure
    can mean a compromised machine remains trusted. A monitoring dashboard
    transforms Keylime from a security tool into a \textbf{security operations platform}.
  \end{alertblock}

  \vspace{0.2cm}

  \textbf{Key Requirements for Enterprise Adoption:}
  \begin{enumerate}
    \item \textbf{Visibility}: Real-time view of all agent states and attestation results
    \item \textbf{Alerting}: Immediate notification on policy violations and failures
    \item \textbf{Operations}: Manage agents, policies, and certificates from a single pane
    \item \textbf{Compliance}: Audit trail and historical attestation records
    \item \textbf{Performance}: Monitor system health and resource utilization
    \item \textbf{Integration}: Connect with existing SIEM and monitoring tools
  \end{enumerate}

  \vspace{0.2cm}

  \begin{exampleblock}{Target Users}
    \scriptsize
    Security Operations (SecOps) teams, System Administrators, Compliance Officers, and DevSecOps engineers
  \end{exampleblock}
\end{frame}
