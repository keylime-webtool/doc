\begin{frame}{Certificate Management - Overview}
  \begin{block}{Certificate Landscape in Keylime}
    \scriptsize
    Keylime relies on multiple certificate types for identity, authentication, and secure
    communication. The dashboard provides a unified view of all certificates.
  \end{block}

  \vspace{0.1cm}

  \begin{center}

    \end{tikzpicture}
  \end{center}

  \vspace{0.1cm}

  \begin{alertblock}{Certificate Expiry is a Common Outage Cause}
    \scriptsize
    Expired certificates silently break mTLS connections, causing agents to appear as
    ``unreachable'' rather than reporting clear certificate errors. Proactive monitoring
    prevents these outages.
  \end{alertblock}
\end{frame}

\begin{frame}{Certificate Management - Operations}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Certificate Inspection}
        \scriptsize
        Deep-dive into any certificate's details.
      \end{block}

      \vspace{0.1cm}
      \scriptsize

      \textbf{Certificate Detail View:}
      \begin{itemize}
        \item Subject and Issuer DN
        \item Serial number
        \item Validity period (Not Before/After)
        \item Public key algorithm and size
        \item Signature algorithm
        \item Subject Alternative Names (SANs)
        \item Key Usage / Extended Key Usage
        \item Certificate chain visualization
        \item PEM/DER export options
      \end{itemize}

      \vspace{0.1cm}

      \textbf{EK Certificate Validation:}
      \begin{itemize}
        \item TPM vendor verification
        \item Chain validation against known CAs
        \item TCG EK credential profile check
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Automated Management}
        \scriptsize
        Proactive certificate lifecycle management.
      \end{block}

      \vspace{0.1cm}
      \scriptsize

      \textbf{Renewal Automation:}
      \begin{itemize}
        \item Auto-renewal for Keylime CA certs
        \item Renewal workflow with approval
        \item Batch renewal scheduling
        \item Pre-renewal validation
        \item Rollback on renewal failure
      \end{itemize}

      \vspace{0.1cm}

      \textbf{Alert Configuration:}
      \begin{itemize}
        \item 90-day warning (informational)
        \item 30-day warning (action required)
        \item 7-day critical alert
        \item 1-day emergency alert
        \item Expired certificate alert
        \item Custom threshold per cert type
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
