\begin{frame}{Revocation - Alert Workflow}
  \begin{block}{Alert Lifecycle Management}
    \scriptsize
    Each revocation event follows a structured lifecycle that operators can track
    and manage through the dashboard.
  \end{block}

  \vspace{0.1cm}

  \begin{center}
    \begin{tikzpicture}[scale=0.72, transform shape,
        state/.style={rectangle, draw, rounded corners, fill=keylimeblue!15, text width=2cm, text centered, minimum height=1cm, font=\tiny\bfseries},
        arrow/.style={->, >=stealth, thick, font=\tiny}]

      \node[state, fill=dangered!20] (new) at (0,0) {New\\Alert};
      \node[state, fill=warningyellow!20] (ack) at (3.5,0) {Acknowledged};
      \node[state, fill=infoblue!15] (invest) at (7,0) {Under\\Investigation};
      \node[state, fill=successgreen!20] (resolved) at (10.5,0.8) {Resolved};
      \node[state, fill=keylimegray!20] (dismissed) at (10.5,-0.8) {Dismissed};

      \draw[arrow] (new) -- node[above] {operator ack} (ack);
      \draw[arrow] (ack) -- node[above] {assigned} (invest);
      \draw[arrow] (invest) -- node[above] {fixed} (resolved);
      \draw[arrow] (invest) -- node[below] {false positive} (dismissed);
    \end{tikzpicture}
  \end{center}

  \vspace{0.2cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Operator Actions:}
      \begin{itemize}
        \item Acknowledge alert
        \item Assign to team member
        \item Add investigation notes
        \item Reactivate agent after fix
        \item Update policy if false positive
        \item Escalate to security team
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Automated Actions:}
      \begin{itemize}
        \item Auto-escalate after SLA timeout
        \item Group related alerts by correlation
        \item Auto-resolve on successful re-attestation
        \item Notification deduplication
        \item Severity auto-adjustment
      \end{itemize}

      \vspace{0.1cm}
      \textbf{Webhook Security:}
      \begin{itemize}
        \item SSRF protection: validate webhook URLs against allowlist of permitted hosts/CIDRs
        \item Block internal IPs (RFC 1918, link-local)
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
