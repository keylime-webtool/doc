\section{Secret Management \& Transport Security}

\begin{frame}{Secret Management - Credential Lifecycle}
  \begin{block}{Dashboard Secret Inventory}
    \scriptsize
    The dashboard backend manages multiple credentials that must be securely stored,
    rotated, and audited throughout their lifecycle.
  \end{block}

  \vspace{0.1cm}

  \scriptsize
  \begin{center}
    \begin{tabular}{|l|p{2.5cm}|p{2.5cm}|p{2.5cm}|}
      \hline
      \textbf{Secret} & \textbf{Purpose} & \textbf{Storage} & \textbf{Rotation} \\
      \hline
      \hline
      mTLS private key & Keylime API auth & HSM or HashiCorp Vault & Annual + on compromise \\
      \hline
      mTLS certificate & Keylime API auth & File (non-secret) & Before expiry \\
      \hline
      OIDC client secret & User authentication & Vault / K8s Secrets & Quarterly \\
      \hline
      Redis password & Cache access & Vault / K8s Secrets & Quarterly \\
      \hline
      TimescaleDB creds & Metrics storage & Vault / K8s Secrets & Quarterly \\
      \hline
      AMQP broker URL & Task queue auth & Vault / K8s Secrets & Quarterly \\
      \hline
      Webhook tokens & Notification auth & Vault / K8s Secrets & On demand \\
      \hline
    \end{tabular}
  \end{center}

  \vspace{0.1cm}

  \begin{alertblock}{mTLS Private Key Protection}
    \scriptsize
    The mTLS private key grants full Keylime admin access. It must \textbf{never} be
    stored on disk in cleartext. Supported backends: PKCS\#11 HSM, HashiCorp Vault
    Transit, Kubernetes CSI Secret Store, or encrypted file with passphrase
    (development only). Rust's \texttt{rustls} supports custom \texttt{SigningKey}
    trait for HSM/PKCS\#11 offload without OpenSSL wrappers.
  \end{alertblock}
\end{frame}

\begin{frame}{Transport Security - mTLS Sidecar Option}
  \begin{block}{Defense-in-Depth: Envoy/Ghostunnel Sidecar}
    \scriptsize
    As an alternative to in-process PKCS\#11, a TLS-terminating sidecar proxy
    can hold the mTLS credentials, isolating the signing material from the
    application process entirely.
  \end{block}

  \vspace{0.1cm}

  \begin{center}
    \begin{tikzpicture}[scale=0.65, transform shape,
        comp/.style={rectangle, draw, rounded corners, fill=keylimeblue!15, text width=2cm, align=center, minimum height=0.8cm, font=\tiny\bfseries},
        arrow/.style={->, >=stealth, thick}]

      \node[comp, fill=warningyellow!15] (backend) at (0,0) {Dashboard\\Backend};
      \node[comp, fill=codepurple!10] (sidecar) at (5,0) {Envoy / Ghostunnel\\(mTLS sidecar)};
      \node[comp] (verifier) at (10,0.7) {Verifier};
      \node[comp] (registrar) at (10,-0.7) {Registrar};

      \draw[arrow, successgreen] (backend) -- node[above, font=\tiny] {plaintext (loopback)} (sidecar);
      \draw[arrow, keylimeblue] (sidecar) -- node[above, font=\tiny] {mTLS} (verifier);
      \draw[arrow, keylimeblue] (sidecar) -- node[below, font=\tiny] {mTLS} (registrar);
    \end{tikzpicture}
  \end{center}

  \vspace{0.1cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Sidecar Advantages:}
      \begin{itemize}
        \setlength{\itemsep}{0pt}
        \item Private key never enters application memory
        \item Sidecar runs as separate process/container
        \item Envoy supports PKCS\#11, SDS, Vault
        \item Backend talks plaintext to localhost only
        \item Independent security audit scope
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Tenant-Scoped Certificates:}
      \begin{itemize}
        \setlength{\itemsep}{0pt}
        \item Single admin cert is a tenant-escape risk
        \item Issue per-tenant scoped certificates
        \item Sidecar selects cert based on tenant header
        \item Keylime cert scoping (if supported upstream)
        \item Fallback: Envoy route-level mTLS config
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Transport Security - Encrypted Data Paths}
  \begin{block}{All Hops Encrypted}
    \scriptsize
    Every network connection between dashboard components uses TLS encryption.
    No cleartext data paths exist in the production deployment.
  \end{block}

  \vspace{0.05cm}

  \begin{exampleblock}{Encryption at Rest}
    \scriptsize
    TimescaleDB: TDE or filesystem encryption (LUKS). Redis: encrypted RDB/AOF or
    ephemeral mode (no persistence). Dashboard config: Vault-backed, never plaintext YAML.
  \end{exampleblock}
\end{frame}
