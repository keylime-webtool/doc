\begin{frame}{Policy Management - Two-Person Rule}
  \begin{block}{Multi-Party Authorization for Policy Changes}
    \scriptsize
    The dashboard holds an admin mTLS cert with write access to the Verifier.
    A compromised admin session could push a malicious IMA policy that whitelists
    malware. Policy changes require \textbf{two-person approval} (N-of-M quorum).
  \end{block}

  \vspace{0.1cm}

  \begin{center}
    \begin{tikzpicture}[scale=0.68, transform shape,
        state/.style={rectangle, draw, rounded corners, fill=keylimeblue!15, text width=2cm, align=center, minimum height=0.9cm, font=\tiny\bfseries},
        arrow/.style={->, >=stealth, thick, font=\tiny}]

      \node[state, fill=infoblue!15] (draft) at (0,0) {Draft\\(Admin A)};
      \node[state, fill=warningyellow!20] (review) at (3.5,0) {Pending\\Review};
      \node[state, fill=keylimegreen!20] (approved) at (7,0) {Approved\\(Admin B)};
      \node[state, fill=successgreen!20] (applied) at (10.5,0) {Applied to\\Verifier};

      \draw[arrow] (draft) -- node[above] {submit} (review);
      \draw[arrow] (review) -- node[above] {approve} (approved);
      \draw[arrow] (approved) -- node[above] {auto-push} (applied);
      \draw[arrow, dashed, dangered] (review) to[bend right=30] node[below] {reject} (draft);
    \end{tikzpicture}
  \end{center}

  \vspace{0.1cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Approval Workflow:}
      \begin{itemize}
        \setlength{\itemsep}{0pt}
        \item Admin A drafts policy change
        \item System runs impact analysis automatically
        \item Admin B (different user) reviews and approves
        \item Configurable quorum: 2-of-3, 3-of-5, etc.
        \item Approver cannot be the same as drafter
        \item Time-limited approval window (24h default)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Change Management Integration:}
      \begin{itemize}
        \setlength{\itemsep}{0pt}
        \item Optionally gate on ServiceNow Change Request
        \item Policy push blocked until CR approved
        \item CR number linked in audit log
        \item Jira ticket auto-created for review
        \item Emergency bypass with break-glass audit
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
