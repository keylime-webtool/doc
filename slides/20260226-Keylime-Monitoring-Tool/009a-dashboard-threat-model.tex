\section{Dashboard Threat Model}

\begin{frame}{Threat Model - Attack Surface}
  \begin{block}{The Dashboard as a High-Value Target}
    \scriptsize
    The monitoring dashboard holds an admin mTLS certificate with full Keylime API access.
    A compromise of the dashboard is a compromise of the entire attestation infrastructure.
  \end{block}

\begin{frame}{Threat Model - Threat Catalog \& Mitigations}
  \begin{block}{Identified Threats}
    \scriptsize
    Each threat targets a specific component and requires dedicated mitigations.
    Three are rated Critical due to potential for full infrastructure compromise.
  \end{block}

  \vspace{0.2cm}

  \scriptsize
  \begin{center}
    \begin{tabular}{|l|p{3.5cm}|l|p{3cm}|}
      \hline
      \textbf{Threat} & \textbf{Impact} & \textbf{Severity} & \textbf{Mitigation} \\
      \hline
      \hline
      T1: Session Hijack & Unauthorized dashboard access & \textcolor{dangered}{Critical} & OIDC + MFA, short-lived tokens \\
      \hline
      T2: XSS/Injection & Data exfil, admin impersonation & \textcolor{dangered}{Critical} & CSP headers, input sanitization \\
      \hline
      T3: Key Exfiltration & Full Keylime admin access & \textcolor{dangered}{Critical} & HSM/Vault, no disk storage \\
      \hline
      T4: Cache Poisoning & Stale/false agent states shown & \textcolor{warningyellow}{High} & Signed cache entries, TTLs \\
      \hline
      T5: Privilege Abuse & Delete agents, modify policies & \textcolor{warningyellow}{High} & Read-only proxy, audit log \\
      \hline
    \end{tabular}
  \end{center}

  \vspace{0.2cm}

  \begin{alertblock}{Key Principle}
    \scriptsize
    The dashboard must be treated as a \textbf{privileged security component}, not a
    convenience tool. Its compromise surface must be minimized through defense-in-depth:
    mTLS sidecar isolation, two-person policy approval, and tamper-evident audit logging.
  \end{alertblock}
\end{frame}

\begin{frame}{Threat Model - Data Classification}
  \begin{block}{Sensitive Data Handling}
    \scriptsize
    The dashboard must classify, protect, and minimize retention of security-sensitive data
    that transits through or is stored in its components.
  \end{block}

  \vspace{0.1cm}

  \tiny
  \begin{center}
    \begin{tabular}{|p{2.5cm}|l|p{2.5cm}|p{3cm}|}
      \hline
      \textbf{Data Type} & \textbf{Classification} & \textbf{Storage Policy} & \textbf{Protection} \\
      \hline
      \hline
      mTLS private key & \textcolor{dangered}{SECRET} & Never on disk & HSM or Vault only \\
      \hline
      EK/AK public keys & \textcolor{warningyellow}{CONFIDENTIAL} & Cache only (TTL) & Encrypt at rest \\
      \hline
      TPM quotes & \textcolor{warningyellow}{CONFIDENTIAL} & Do not cache & Pass-through only \\
      \hline
      IMA policy content & \textcolor{warningyellow}{CONFIDENTIAL} & Cache with TTL & Encrypt at rest \\
      \hline
      Agent IPs/UUIDs & \textcolor{infoblue}{INTERNAL} & TimeSeries DB & Encrypt at rest \\
      \hline
      Attestation results & \textcolor{infoblue}{INTERNAL} & TimeSeries DB & Encrypt at rest \\
      \hline
      Dashboard audit log & \textcolor{warningyellow}{CONFIDENTIAL} & Append-only store & Hash-chained, signed \\
      \hline
      Dashboard credentials & \textcolor{dangered}{SECRET} & Never persisted & OIDC session tokens \\
      \hline
    \end{tabular}
  \end{center}

  \vspace{0.1cm}

  \begin{alertblock}{Data Minimization Principle}
    \scriptsize
    The dashboard must \textbf{never cache or store} TPM quotes, IMA measurement logs,
    or raw boot event logs. These are passed through from the Keylime API to the UI
    on demand and discarded. Only attestation \textit{results} (pass/fail + timestamp)
    are retained for historical analysis.
  \end{alertblock}
\end{frame}
