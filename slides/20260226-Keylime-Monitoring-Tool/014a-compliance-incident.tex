\begin{frame}{Compliance - Tamper-Evident Audit Logging}
  \begin{block}{Audit Log Integrity}
    \scriptsize
    The dashboard's own audit log must be tamper-evident to serve as
    valid compliance evidence. Self-reported logs stored in a mutable database
    are insufficient for SOC 2 or FedRAMP auditors.
  \end{block}

  \vspace{0.2cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Hash-Chain Integrity:}
      \begin{itemize}
        \item Each log entry includes SHA-256 hash of previous entry
        \item Chain root anchored to external timestamp (RFC 3161)
        \item Periodic chain checkpoints to Rekor transparency log
        \item Tamper detection on startup and periodic verification
      \end{itemize}

      \vspace{0.15cm}

      \textbf{Immutable Storage Options:}
      \begin{itemize}
        \item Append-only database tables
        \item S3 Object Lock (WORM)
        \item Rekor transparency log
        \item Syslog forwarding (real-time)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Log Entry Structure:}
      \begin{itemize}
        \item Timestamp (UTC, millisecond precision)
        \item Actor (user identity from OIDC)
        \item Action (CRUD operation + target)
        \item Resource (agent ID, policy name)
        \item Source IP and user agent
        \item Result (success/denied/error)
        \item Previous entry hash (chain link)
      \end{itemize}

      \vspace{0.15cm}

      \textbf{Retention Policy:}
      \begin{itemize}
        \item Configurable per deployment
        \item Minimum 1 year for compliance
        \item Archival to cold storage (S3/NFS)
        \item Secure deletion after retention
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Incident Response - Integration}
  \begin{block}{Security Incident Workflow Integration}
    \scriptsize
    Attestation failures are potential security incidents that must integrate
    with existing enterprise incident response tooling and processes.
  \end{block}

  \vspace{0.1cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Ticketing Integration:}
      \begin{itemize}
        \setlength{\itemsep}{0pt}
        \item ServiceNow incident + change request
        \item Jira issue auto-creation
        \item PagerDuty alert routing
        \item OpsGenie on-call escalation
        \item Bidirectional status sync
      \end{itemize}

      \vspace{0.1cm}

      \textbf{Change Management (ITSM):}
      \begin{itemize}
        \setlength{\itemsep}{0pt}
        \item Policy changes gated on approved CR
        \item ServiceNow CR auto-creation on policy draft
        \item CR number linked in dashboard audit log
        \item Emergency bypass with break-glass audit
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Automated Remediation:}
      \begin{itemize}
        \item Network quarantine via NAC integration
        \item Agent isolation (stop attestation)
        \item Trigger re-provisioning workflow
        \item Configurable playbooks per failure type
        \item Manual approval gate for destructive actions
      \end{itemize}

      \vspace{0.1cm}

      \textbf{SIEM Integration (Day 1):}
      \begin{itemize}
        \item Syslog (CEF/LEEF format)
        \item Splunk HEC (HTTP Event Collector)
        \item Elastic Common Schema (ECS)
        \item Prometheus metrics endpoint
        \item OpenTelemetry traces
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
