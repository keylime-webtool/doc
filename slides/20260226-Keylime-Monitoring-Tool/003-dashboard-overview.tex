\begin{frame}{Dashboard - Key Performance Indicators}
  \begin{block}{Top-Level KPIs}
    \scriptsize
    The main dashboard presents at-a-glance health metrics computed from
    Keylime's Verifier and Registrar APIs.
  \end{block}

  \vspace{0.1cm}

  \scriptsize
  \begin{center}
    \begin{tabular}{|l|p{4cm}|p{3.5cm}|}
      \hline
      \textbf{KPI} & \textbf{Source} & \textbf{Alert Threshold} \\
      \hline
      \hline
      Total Active Agents & Verifier: \texttt{GET /agents/} & N/A (informational) \\
      \hline
      Failed Agents & Agents in state 7, 9, 10 & Any count $>$ 0 \\
      \hline
      Attestation Success Rate & Computed from attestation history & Below 99\% \\
      \hline
      Avg Attestation Latency & Time between quote requests & Above 2x quote\_interval \\
      \hline
      Certificate Expiry Warnings & mTLS cert validity check & Within 30 days \\
      \hline
      Active IMA Policies & Verifier: \texttt{GET /policies/ima} & N/A (informational) \\
      \hline
      Revocation Events (24h) & Revocation notifier events & Any count $>$ 0 \\
      \hline
      Consecutive Failures & Per-agent failure counter & Above 3 consecutive \\
      \hline
      Registration Count & Registrar: \texttt{GET /agents/} & Unexpected re-registrations \\
      \hline
    \end{tabular}
  \end{center}

  \vspace{0.2cm}

  \begin{alertblock}{Data Refresh}
    \scriptsize
    KPIs refresh every 30 seconds via polling or WebSocket push.
    Alert thresholds are configurable per deployment.
  \end{alertblock}
\end{frame}

\begin{frame}{Dashboard - Navigation Structure}
  \begin{columns}[T]
    \begin{column}{0.35\textwidth}
      \textbf{Sidebar Navigation:}
      \scriptsize
      \begin{enumerate}
        \item \textbf{Dashboard} - Fleet overview
        \item \textbf{Agents} - Fleet management
        \item \textbf{Attestations} - Analytics
        \item \textbf{Policies} - IMA \& MB
        \item \textbf{Certificates} - TLS/TPM certs
        \item \textbf{Revocations} - Alert history
        \item \textbf{Performance} - System metrics
        \item \textbf{Audit Log} - Security events
        \item \textbf{Integrations} - Backend status
        \item \textbf{Settings} - Configuration
      \end{enumerate}
    \end{column}
    \begin{column}{0.6\textwidth}
      \textbf{Common UI Elements:}
      \scriptsize
      \begin{itemize}
        \item \textbf{Search Bar}: Global agent search by UUID, IP, or hostname
        \item \textbf{Time Range Selector}: Filter data by 1h, 6h, 24h, 7d, 30d, custom
        \item \textbf{Auto-Refresh Toggle}: Enable/disable live updates
        \item \textbf{Export}: CSV/JSON export for compliance reporting
        \item \textbf{Dark/Light Mode}: Theme preference
        \item \textbf{Role-Based Access}: Admin, Operator, Viewer roles
      \end{itemize}

      \vspace{0.2cm}

      \textbf{Notification System:}
      \begin{itemize}
        \item In-app notification bell with badge count
        \item Email/Slack integration for critical alerts
        \item Configurable severity thresholds
        \item Alert acknowledgment workflow
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
