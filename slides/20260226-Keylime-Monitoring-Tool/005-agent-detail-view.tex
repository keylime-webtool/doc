\begin{frame}{Agent Detail - Tabs and Sub-Views (1/2)}
  \begin{block}{Deep-Dive Tabs}
    \scriptsize
    Each agent detail page provides six specialized tabs for deep investigation.
  \end{block}

  \vspace{0.2cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{1. Timeline Tab:}
      \begin{itemize}
        \item Attestation success/failure history
        \item State transition events
        \item Timestamped event log
        \item Zoomable time range
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{2. PCR Values Tab:}
      \begin{itemize}
        \item Current PCR bank values (SHA-256)
        \item PCR change history with diffs
        \item Expected vs. actual comparison
        \item PCR policy violations highlighted
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.3cm}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{3. IMA Log Tab:}
      \begin{itemize}
        \item IMA measurement list entries
        \item File hash verification status
        \item Policy match/mismatch indicators
        \item Search by file path or hash
        \item \texttt{next\_ima\_ml\_entry} progress
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{4. Boot Log Tab:}
      \begin{itemize}
        \item UEFI event log entries
        \item Measured boot sequence validation
        \item Boot policy compliance status
        \item Secure Boot state indicators
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

  \vspace{0.3cm}

  \begin{alertblock}{Cross-Tab Navigation}
    \scriptsize
    Tabs are interconnected -- clicking a PCR value in the \textbf{PCR Values} tab
    can jump to the corresponding IMA entries in the \textbf{IMA Log} tab, and
    certificate warnings link directly to the \textbf{Certificates} tab for renewal.
  \end{alertblock}
\end{frame}
