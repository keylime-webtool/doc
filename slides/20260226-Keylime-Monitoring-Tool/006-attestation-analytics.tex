\begin{frame}{Attestation Analytics - Automatic Correlation}
  \begin{exampleblock}{Automatic Correlation}
    \scriptsize
    Dashboard correlates failures across agents -- if multiple agents fail simultaneously
    with the same reason, it may indicate a policy update issue rather than actual compromise.
  \end{exampleblock}

  \vspace{0.3cm}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Correlation Rules}
        \scriptsize
        \begin{itemize}
          \item \textbf{Temporal}: Multiple agents failing within the same time window
          \item \textbf{Causal}: Same failure reason across agents (e.g., same IMA hash mismatch)
          \item \textbf{Topological}: Failures grouped by datacenter, subnet, or verifier
          \item \textbf{Policy-linked}: Failures matching a recent policy update
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Correlation Outcomes}
        \scriptsize
        \begin{itemize}
          \item \textbf{Mass policy update}: Many agents fail after policy change -- likely a policy misconfiguration
          \item \textbf{Infrastructure event}: Cluster of timeouts -- likely a network or infrastructure issue
          \item \textbf{Targeted attack}: Single agent with unique failure -- warrants immediate investigation
          \item \textbf{Software rollout}: PCR/IMA changes after a package update wave
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Attestation Analytics - Actionable Insights}
  \begin{alertblock}{Actionable Insight}
    \scriptsize
    Correlated failures are grouped into a single incident with a suggested root cause,
    reducing alert fatigue and enabling faster resolution.
  \end{alertblock}

  \vspace{0.3cm}

  \begin{block}{Incident Grouping Example}
    \scriptsize
    \begin{itemize}
      \item 15 agents report \textcolor{dangered}{IMA policy violation} within a 2-minute window
      \item All 15 agents share the same IMA policy: \texttt{production-v2}
      \item Policy \texttt{production-v2} was updated 3 minutes ago
      \item \textbf{Suggested root cause}: Policy update introduced a missing hash entry
      \item \textbf{Recommended action}: Review latest policy change and consider rollback
    \end{itemize}
  \end{block}

  \vspace{0.3cm}

  \begin{exampleblock}{Benefits}
    \scriptsize
    \begin{itemize}
      \item Reduces 15 individual alerts to 1 correlated incident
      \item Automatically links to the triggering policy change
      \item Provides one-click rollback option from the incident view
    \end{itemize}
  \end{exampleblock}
\end{frame}
