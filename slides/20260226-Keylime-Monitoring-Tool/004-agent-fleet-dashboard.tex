\begin{frame}{Agent Fleet - Filtering and Search}
  \begin{block}{Advanced Filtering Capabilities}
    \scriptsize
    Administrators can filter the agent fleet using multiple criteria simultaneously.
  \end{block}

  \vspace{0.2cm}

  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Filter Criteria:}
      \begin{itemize}
        \item \textbf{Agent UUID}: Exact or partial match
        \item \textbf{IP Address}: CIDR range support
        \item \textbf{Operational State}: Multi-select dropdown
        \item \textbf{Verifier Assignment}: Which verifier instance
        \item \textbf{IMA Policy}: Filter by assigned policy
        \item \textbf{MB Policy}: Filter by boot policy
        \item \textbf{Last Attestation}: Time range filter
        \item \textbf{Failure Count}: Min/max threshold
        \item \textbf{Registration Date}: Date range
        \item \textbf{API Version}: Agent-supported version
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Bulk Operations:}
      \begin{itemize}
        \item Select multiple agents (checkbox)
        \item \textbf{Reactivate}: Resume monitoring for selected
        \item \textbf{Stop}: Pause monitoring for selected
        \item \textbf{Delete}: Remove from verifier
        \item \textbf{Reassign Policy}: Change IMA/MB policy
        \item \textbf{Export}: Download agent data as CSV
      \end{itemize}

      \vspace{0.2cm}

      \textbf{Sort Options:}
      \begin{itemize}
        \item By state (failures first)
        \item By last attestation time
        \item By failure count (descending)
        \item By registration date
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
